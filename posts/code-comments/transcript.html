// Function to estimate the value of pi using the Monte Carlo method
function estimate_pi(steps=1000000) {
  var inside = 0;
  var outside = 0;
  var step = 0;
  
  // Loop through the specified number of steps
  while (step < steps) {
     x = (Math.random()-0.5)*2;
     y = (Math.random()-0.5)*2;
     
     // Check if the point (x, y) is inside the unit circle
     if (Math.sqrt(x*x+y*y)<=1) {
       inside+=1;
     } else {
       outside+=1;
     }
     step+=1;
  }
  
  // Return an estimate of pi using the ratio of points inside the circle to total points
  return 4*(inside)/(inside+outside);
}

var d = new Date();
start_time = d.getTime();

// Call the estimate_pi function with a specified number of steps
pi = estimate_pi(10000000);
console.log("π ≈ "+pi);

var d2 = new Date();
end_time = d2.getTime();

// Calculate and log the elapsed time in seconds
console.log("Elapsed time: "+(end_time-start_time)/1000+" sec");
